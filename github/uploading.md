# github 업로드하는 법

  1. git config --global user.name "xx"

  2. git config --global user.email "xxxxxx@gmail.com"

  3. git init

  4. git status

  5. git add.

  6. git commit -m "변경내용"

  7. git remote add origin https://github.com/username/xxx.git

  8. git remote -v

  9. git push -u origin master
